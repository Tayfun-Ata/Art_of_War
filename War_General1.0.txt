import pygame
import sys

# Initialize Pygame
pygame.init()

# Screen dimensions
SCREEN_WIDTH = 800
SCREEN_HEIGHT = 600
GRID_SIZE = 50

# Colors
WHITE = (255, 255, 255)
BLACK = (0, 0, 0)
RED = (255, 0, 0)
GREEN = (0, 255, 0)

# Create the screen
screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT))
pygame.display.set_caption("Simple Strategy Game")

# Unit class
class Unit:
    def __init__(self, x, y, color):
        self.x = x
        self.y = y
        self.color = color

    def draw(self):
        pygame.draw.rect(screen, self.color, (self.x * GRID_SIZE, self.y * GRID_SIZE, GRID_SIZE, GRID_SIZE))

    def move(self, dx, dy):
        self.x += dx
        self.y += dy

# Create units
player_unit = Unit(5, 5, GREEN)
enemy_unit = Unit(10, 10, RED)

# Main game loop
def main():
    clock = pygame.time.Clock()

    while True:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                pygame.quit()
                sys.exit()
            elif event.type == pygame.KEYDOWN:
                if event.key == pygame.K_LEFT:
                    player_unit.move(-1, 0)
                elif event.key == pygame.K_RIGHT:
                    player_unit.move(1, 0)
                elif event.key == pygame.K_UP:
                    player_unit.move(0, -1)
                elif event.key == pygame.K_DOWN:
                    player_unit.move(0, 1)

        # Clear the screen
        screen.fill(WHITE)

        # Draw the grid
        for x in range(0, SCREEN_WIDTH, GRID_SIZE):
            pygame.draw.line(screen, BLACK, (x, 0), (x, SCREEN_HEIGHT))
        for y in range(0, SCREEN_HEIGHT, GRID_SIZE):
            pygame.draw.line(screen, BLACK, (0, y), (SCREEN_WIDTH, y))

        # Draw units
        player_unit.draw()
        enemy_unit.draw()

        # Update the display
        pygame.display.flip()
        clock.tick(30)

if __name__ == "__main__":
    main()
